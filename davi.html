(mantenha o CSS e HTML igual, substitua apenas o script) const Game = { canvas: document.getElementById('canvas'), ctx: null, score: 0, blocks: [], fallingBlock: null, gameActive: false, cameraY: 0, targetCameraY: 0, towerLean: 0, leanVelocity: 0, swing: { angle: 0, speed: 0.012, x: 0 }, blockW: 140, blockH: 45, dropDist: 250, init() { this.ctx = this.canvas.getContext('2d'); this.resize(); window.addEventListener('resize', () => this.resize()); this.canvas.addEventListener('pointerdown', (e) => { e.preventDefault(); this.launch(); }); this.render(); }, resize() { this.width = this.canvas.width = this.canvas.offsetWidth; this.height = this.canvas.height = this.canvas.offsetHeight; }, start() { this.score = 0; this.cameraY = 0; this.targetCameraY = 0; this.towerLean = 0; this.leanVelocity = 0; this.fallingBlock = null; // O primeiro bloco é o centro real (base fixa) this.blocks = [{ x: this.width/2 - 70, y: this.height - 80 }]; this.gameActive = true; document.getElementById('overlay').style.display = 'none'; document.getElementById('score').innerText = "0"; }, launch() { if(!this.gameActive || this.fallingBlock) return; // Calculamos a posição de onde o bloco solta // Para não bugar com a câmera, usamos o Y relativo ao último bloco this.fallingBlock = { x: this.swing.x - this.blockW / 2, y: (this.blocks[this.blocks.length-1].y - this.dropDist), vy: 0, gravity: 0.7 }; }, processLanding() { const lastBlock = this.blocks[this.blocks.length - 1]; // IMPORTANTE: Para fixar sem mudar de posição, calculamos o desvio real // levando em conta que a torre está inclinada. const diff = this.fallingBlock.x - lastBlock.x; if (Math.abs(diff) < this.blockW) { this.score++; // Adiciona o bloco exatamente onde ele caiu this.blocks.push({ x: this.fallingBlock.x, y: lastBlock.y - this.blockH }); // Física Gradual: Mais longe do centro = mais força de queda this.leanVelocity += (diff * 0.006); if (this.blocks.length > 4) { this.targetCameraY += this.blockH; } document.getElementById('score').innerText = this.score; } else { this.gameOver(); } this.fallingBlock = null; }, gameOver() { this.gameActive = false; const overlay = document.getElementById('overlay'); overlay.style.display = 'flex'; overlay.innerHTML = `

A TORRE CAIU!

Sua obra alcançou ${this.score} níveis.

`; }, drawBlock(b, isGhost = false) { const ctx = this.ctx; ctx.save(); if(isGhost) ctx.globalAlpha = 0.5; let grad = ctx.createLinearGradient(b.x, b.y, b.x, b.y + this.blockH); grad.addColorStop(0, "#FFD700"); grad.addColorStop(1, "#B8860B"); ctx.fillStyle = grad; ctx.strokeStyle = "#FFF9C4"; ctx.lineWidth = 2; ctx.beginPath(); ctx.roundRect(b.x, b.y, this.blockW, this.blockH, 4); ctx.fill(); ctx.stroke(); ctx.restore(); }, render() { const ctx = this.ctx; // Limpeza total para evitar rastros e problemas de fundo ctx.clearRect(0, 0, this.width, this.height); this.cameraY += (this.targetCameraY - this.cameraY) * 0.08; // --- MUNDO COM ROTAÇÃO --- ctx.save(); ctx.translate(0, this.cameraY); const baseY = this.height - 80; const baseX = this.width / 2; if (this.gameActive && this.blocks.length > 1) { this.towerLean += this.leanVelocity; // Gravidade da inclinação (aumenta quanto mais torto está) this.leanVelocity += (this.towerLean * 0.0009); } // Ponto de pivot no centro da base ctx.translate(baseX, baseY); ctx.rotate(this.towerLean * Math.PI / 180); ctx.translate(-baseX, -baseY); // Desenha todos os blocos já fixados this.blocks.forEach(b => this.drawBlock(b)); // Queda do bloco atual if(this.fallingBlock) { this.fallingBlock.vy += this.fallingBlock.gravity; this.fallingBlock.y += this.fallingBlock.vy; this.drawBlock(this.fallingBlock); const targetY = this.blocks[this.blocks.length-1].y - this.blockH; if(this.fallingBlock.y >= targetY) { this.fallingBlock.y = targetY; this.processLanding(); } } ctx.restore(); // --- INTERFACE E ELEMENTOS FIXOS (CORDA) --- if (this.gameActive && !this.fallingBlock) { this.swing.angle += this.swing.speed; // Oscilação mais suave this.swing.x = (this.width/2) + Math.sin(this.swing.angle) * (this.width * 0.38); // Posição visual da mira (compensando a câmera) const swingY = (this.blocks[this.blocks.length-1].y - this.dropDist); // Desenha a corda vindo do topo da tela (fixa) ctx.beginPath(); ctx.moveTo(this.width/2, -100); ctx.lineTo(this.swing.x, swingY + this.cameraY); ctx.strokeStyle = "rgba(255,215,0,0.4)"; ctx.lineWidth = 2; ctx.stroke(); // Desenha o bloco fantasma na corda ctx.save(); ctx.translate(0, this.cameraY); this.drawBlock({ x: this.swing.x - this.blockW/2, y: swingY }, true); ctx.restore(); } // Fim de jogo se inclinar demais if (Math.abs(this.towerLean) > 28) { this.gameOver(); } requestAnimationFrame(() => this.render()); } }; Game.init();
